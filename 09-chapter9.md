func.genMod7 <- function() {
  x <- 8
  while(x %% 7 != 0) {
    x <- sample(100:200, size = 1)
  }
  return(x)
}

func.genVectorMod7 <- function(size) {
  v <- c()
  
  for(i in 1:size) {
    v <- c(v, func.genMod7())

vector.info <- function(vector) {
  x <- list()
  x$Sum <- sum(vector)
  x$Mean <- mean(vector)
  x$Median <- median(vector)
  x$Min <- min(vector)
  x$Max <- max(vector)
  return(x)
}
